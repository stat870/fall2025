8.1 Splines are special cases of non-parametric statistics

We can generally describe smoothing splines as: \[f(x) = \sum_{i=1}^k B_i^m(x) \beta_i \]

8.2.1 Example

x <- seq(1, 50, by = 1)
y <- 15 + 10*(.5*sin(.3*x)+.1*cos(x)) + rnorm(length(x), 0, 1.5)
plot(x, y)

splines_df <- data.frame(x, y)
m_spline_tp <- gam(y ~ s(x, bs = "tp"))
m_spline_bs <- gam(y ~ s(x, bs = "bs"))
m_spline_tpresid <- gam(list(y ~ 1, ~s(x, bs = "tp")), 
                        family = gaulss(),
                        data = splines_df)

splines_df$tp_spline <- predict(m_spline_tp, type = "response")
splines_df$tp_spline_se <- predict(m_spline_tp, type = "response", se.fit = T)$se.fit
splines_df$bs_spline <- predict(m_spline_bs, type = "response")

splines_df %>% 
  ggplot(aes(x, y))+
  theme_classic()+
  theme(panel.border = element_blank(), 
        panel.grid = element_blank())+
  geom_line(aes(y = tp_spline+(tp_spline_se*1.96)), linetype = 2)+
  geom_line(aes(y = tp_spline-(tp_spline_se*1.96)), linetype = 2)+
  geom_line(aes(y = tp_spline))+
  geom_point()

x <- seq(1, 50, by = 1)
xx2 <- seq(1, 50, length.out = length(x))
set.seed(42)
y <- 15 + 15*sin(sqrt(x*.15 + xx2*.25 - x*xx2*.006))+ rnorm(length(x), 0, 1.5)
plot(x, y)

splines_df <- data.frame(x, xx2, y)
m_spline_tp <- gam(y ~ s(x, bs = "tp"), method = "REML", data = splines_df)
m_spline_bs <- gam(y ~ s(x, bs = "bs", k = 7), data = splines_df)
m_spline_tpresid <- gam(list(y ~ 1, ~s(x, bs = "tp")), 
                        family = gaulss(),
                        data = splines_df)

splines_df$tp_spline <- predict(m_spline_tp, type = "response")
splines_df$tp_spline_se <- predict(m_spline_tp, type = "response", se.fit = T)$se.fit
splines_df$bs_spline <- predict(m_spline_bs, type = "response")
splines_df$bs_spline_se <- predict(m_spline_bs, type = "response", se.fit = T)$se.fit

knots <- m_spline_bs$smooth[[1]]$knots

coef(m_spline_bs)[grep("s\\(x\\)", names(coef(m_spline_bs)))]
##   s(x).1   s(x).2   s(x).3   s(x).4   s(x).5   s(x).6 
## 28.94401 30.18179 28.23278 25.05084 16.06937 16.99287
beta_smooth <- coef(m_spline_bs)[grep("s\\(x\\)", names(coef(m_spline_bs)))]

Xp <- predict(m_spline_bs, type = "lpmatrix")  # each column = basis function

# Remove intercept column for plotting
Xp_smooth <- Xp[, grep("s\\(x\\)", colnames(Xp))]

# Plot basis functions
matplot(x, Xp_smooth, type = "l", lty = 1, col = rainbow(ncol(Xp_smooth)),
        main = "Spline Basis Functions", xlab = "x", ylab = "Basis value")

# Add contribution from each basis function
y_basis <- sweep(Xp_smooth, 2, beta_smooth, "*")
matplot(x, y_basis, type = "l", lty = 1, col = rainbow(ncol(Xp_smooth)),
        main = "Contribution of Each Basis Function", xlab = "x", ylab = "Contribution")

df_basis <- as.data.frame(cbind(x, Xp_smooth)) %>% 
  pivot_longer(cols = `s(x).1`:`s(x).6`)
splines_df %>% 
  ggplot(aes(x, y))+  
  geom_line(aes(y = 18 + value*10, 
                group = name, 
                color = name),
            show.legend = F,
            data = df_basis, 
            alpha = .6)+
  theme_classic()+
  theme(panel.border = element_blank(), 
        panel.grid = element_blank())+
  geom_vline(aes(xintercept = knots), data = data.frame(knots), linetype = 2)+
  geom_line(aes(y = bs_spline+(bs_spline_se*1.96)), linetype = 2)+
  geom_line(aes(y = bs_spline-(bs_spline_se*1.96)), linetype = 2)+
  geom_line(aes(y = bs_spline))+
  coord_cartesian(xlim = c(0, 50))+
  labs(x = "Columns (x1)", y = "y or f(x1)")+
  geom_point()

library(tidyverse)
library(mgcv)
library(agridat)

data(burgueno.rowcol)
df <- burgueno.rowcol

m <- lm(yield ~ gen + rep, 
        data = df)

df$res1 <- residuals(m)

df %>% 
  ggplot(aes(col, row))+
  coord_equal()+
  labs(fill = "residual", 
       x = "column", 
       y = "row")+
  scale_fill_viridis_b()+
  geom_tile(aes(fill = res1))

df %>% 
  ggplot(aes(col, row))+
  coord_equal()+
  labs(fill = "residual", 
       x = "column", 
       y = "row")+
  geom_contour_filled(aes(z = res1), 
               bins = 5,
               color = "black", size = 0.5)

library(mgcv)
m_splines <- gam(yield ~ gen + rep + s(row, col, bs = "tp"),
                 data = df)

m_splines2 <- gam(list(yield ~ gen + rep + s(row, col, bs = "tp"), ~s(row, col, bs = "tp")),
                  family = gaulss(),
                  data = df)

m_splines3 <- gam(yield ~ gen + rep + s(row, col, bs = "tp"),
                  data = df)

df$res2 <- residuals(m_splines)
df$res3 <- residuals(m_splines2)
df$res4 <- residuals(m_splines3)


df %>% 
  ggplot(aes(col, row))+
  coord_equal()+
  labs(fill = "residual", 
       x = "column", 
       y = "row")+
  geom_contour_filled(aes(z = res3), 
               bins = 5,
               color = "black", size = 0.5)

plot(m)

plot(m_splines)

plot(m_splines2)
